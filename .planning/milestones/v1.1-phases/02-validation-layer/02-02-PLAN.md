   ```typescript
   const raw = await req.json()
   const parsed = createClienteSchema.safeParse(raw)
   if (!parsed.success) {
     const msg = parsed.error.errors.map(e => e.message).join(', ')
     return NextResponse.json({ data: null, error: msg }, { status: 400 })
   }
   const body = parsed.data
   ```
