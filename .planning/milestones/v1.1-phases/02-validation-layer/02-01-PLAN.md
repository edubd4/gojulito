**`createClienteSchema`** — validates POST body for both dashboard and webhook routes:
```typescript
import { z } from 'zod'
import { ESTADO_CLIENTE, CANAL_INGRESO } from '@/lib/constants'

export const createClienteSchema = z.object({
  nombre: z.string().min(1, 'Nombre es requerido').transform(v => v.trim()),
  telefono: z.string().min(1, 'Telefono es requerido').transform(v => v.trim()),
  email: z.string().email('Email invalido').optional().or(z.literal('')).transform(v => v?.trim() || undefined),
  dni: z.string().optional().transform(v => v?.trim() || undefined),
  fecha_nac: z.string().optional(),
  provincia: z.string().optional().transform(v => v?.trim() || undefined),
  canal: z.enum([CANAL_INGRESO.SEMINARIO, CANAL_INGRESO.WHATSAPP, CANAL_INGRESO.INSTAGRAM, CANAL_INGRESO.REFERIDO, CANAL_INGRESO.CHARLA, CANAL_INGRESO.OTRO], { required_error: 'Canal es requerido' }),
  estado: z.enum([ESTADO_CLIENTE.PROSPECTO, ESTADO_CLIENTE.ACTIVO, ESTADO_CLIENTE.FINALIZADO, ESTADO_CLIENTE.INACTIVO], { required_error: 'Estado es requerido' }),
  grupo_familiar_id: z.string().uuid().optional(),
  observaciones: z.string().optional().transform(v => v?.trim() || undefined),
})
```

**`patchClienteSchema`** — validates PATCH body (all fields optional):
```typescript
export const patchClienteSchema = z.object({
  nombre: z.string().min(1).optional(),
  telefono: z.string().nullable().optional(),
  email: z.string().email().nullable().optional(),
  dni: z.string().nullable().optional(),
  fecha_nac: z.string().nullable().optional(),
  provincia: z.string().nullable().optional(),
  canal: z.enum([...canal values...]).optional(),
  estado: z.enum([...estado values...]).optional(),
  grupo_familiar_id: z.string().uuid().nullable().optional(),
  observaciones: z.string().nullable().optional(),
})
```

Export both. Also export inferred types: `export type CreateClienteInput = z.infer<typeof createClienteSchema>` and `export type PatchClienteInput = z.infer<typeof patchClienteSchema>`.

Use Claude's discretion on error messages (Spanish, concise). Use `.safeParse()` will be done in the route handlers (Plan 02), not here.

```typescript
  fecha_vencimiento_deuda: z.string().nullable().optional(),
  referencia_grupo: z.string().nullable().optional(),
  notas: z.string().nullable().optional(),
})
```

Add `.refine()` to enforce: if `tipo === 'VISA'` then `visa_id` must be present (replicating the existing 422 check).

**`patchPagoSchema`** — validates PATCH body:

```typescript
export const patchPagoSchema = z.object({
```
